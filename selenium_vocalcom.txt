# downloader/vocalcom_downloader.py

import os
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager
from dotenv import load_dotenv

load_dotenv()


class BasePage:
    def __init__(self, driver):
        self.driver = driver
        self.wait = WebDriverWait(driver, 15)
        self.original_tab = None

    def _open_url(self, url):
        self.driver.get(url)

    def _click(self, locator):
        self.wait.until(EC.element_to_be_clickable(locator)).click()

    def _type_text(self, locator, text):
        element = self.wait.until(EC.visibility_of_element_located(locator))
        element.clear()
        element.send_keys(text)

    def switch_to_new_tab(self):
        self.original_tab = self.driver.current_window_handle
        self.wait.until(lambda d: len(d.window_handles) > 1)
        for tab in self.driver.window_handles:
            if tab != self.original_tab:
                self.driver.switch_to.window(tab)
                break
        print(f"URL du nouvel onglet: {self.driver.current_url}")
        return self.driver.current_url

    def switch_to_original_tab(self):
        if self.original_tab:
            self.driver.switch_to.window(self.original_tab)
            print(f"Retour à l'onglet original: {self.driver.current_url}")
        else:
            raise Exception("Aucun onglet original enregistré")


class LoginPage(BasePage):
    def __init__(self, driver):
        super().__init__(driver)
        self.url = "https://tapp1240wv.corp.telma.mg/hermes360/Admin/Launcher/login"
        self.username_input = (By.ID, "usrID")
        self.password_input = (By.ID, "usrPWD")
        self.login_button = (By.CSS_SELECTOR, "Input.send")
        self.username = os.getenv("USER_NAME_VOCALCOM")
        self.password = os.getenv("PASSWORD")
        if not self.username or not self.password:
            raise Exception("Les credentials doivent être définis dans le fichier .env")

    def login(self):
        self._open_url(self.url)
        self._type_text(self.username_input, self.username)
        self._click(self.login_button)
        self._type_text(self.password_input, self.password)
        self._click(self.login_button)


class DashboardPage(BasePage):
    def __init__(self, driver):
        super().__init__(driver)
        self.workspace_button = (By.CSS_SELECTOR, "#hermes .workspace")
        self.reporting_button = (By.CSS_SELECTOR, ".button-panel.reporting .button-name")

    def click_workspace(self):
        self._click(self.workspace_button)

    def click_reporting(self):
        self._click(self.reporting_button)

    def verify_reporting_page(self):
        expected_url = "https://tapp1240wv.corp.telma.mg/hermes360/Admin/Launcher/Start.aspx"
        current_url = self.driver.current_url
        if current_url.startswith(expected_url):
            print("Page de reporting correctement chargée")
            return True
        print(f"URL incorrecte: attendu {expected_url}, obtenu {current_url}")
        return False


class ReportingPage(BasePage):
    def open_saved_reports(self):
        """Clique sur les rapports sauvegardés via JS dans l'iframe"""
        script = """
            let iframe = document.querySelector("iframe#main");
            let iframeDoc = iframe.contentDocument || iframe.contentWindow.document;
            let menuGeneral = iframeDoc.querySelector("#Mnu_General");
            if(menuGeneral) menuGeneral.click();
            let listesRapportsSauvegardes = iframeDoc.querySelector('[onclick="return ShowLoadDialog()"]');
            if(listesRapportsSauvegardes) listesRapportsSauvegardes.click();
        """
        self.driver.execute_script(script)


def download_reports():
    """Fonction principale exportable pour récupérer les rapports Vocalcom"""
    options = webdriver.ChromeOptions()
    options.add_argument("--start-maximized")
    # Désactiver les popups et downloads prompt si nécessaire
    prefs = {
        "download.prompt_for_download": False,
        "download.directory_upgrade": True
    }
    options.add_experimental_option("prefs", prefs)

    driver = webdriver.Chrome(
        service=Service(ChromeDriverManager().install()),
        options=options
    )

    try:
        # 1. Login
        login_page = LoginPage(driver)
        login_page.login()

        # 2. Dashboard
        dashboard = DashboardPage(driver)
        dashboard.click_workspace()
        dashboard.click_reporting()
        dashboard.switch_to_new_tab()
        dashboard.verify_reporting_page()

        # 3. Reporting
        reporting = ReportingPage(driver)
        reporting.open_saved_reports()

    finally:
        driver.quit()


# Pour tester le téléchargement localement
if __name__ == "__main__":
    download_reports()
